\chapter{数据模式与运行配置补充}
\label{app:api}

\section{关系模式（由 \texttt{ensure\_analytics\_ddl} 创建）}
下列表在首次 \texttt{storage->open} 成功时按需创建。主键表名由 \texttt{TableParams::mysql\_table} 指定（默认 \texttt{otsh\_keys}），单列 \texttt{k BIGINT UNSIGNED PRIMARY KEY}，用于持久化当前键集（与内存结构同步策略以 \texttt{/api/init} 的 \texttt{reset} 与插入/删除路径为准）。

\subsection{\texttt{snapshot\_meta}}
为避免 Windows 下 \texttt{lower\_case\_table\_names} 导致列名冲突，规模字段在库中命名为 \texttt{init\_param\_n}、\texttt{table\_n}、\texttt{table\_k}：
\begin{verbatim}
CREATE TABLE IF NOT EXISTS `snapshot_meta`(
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `snapshot_tag` VARCHAR(64),
  `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  `init_param_n` BIGINT NOT NULL,
  `table_n` BIGINT UNSIGNED NOT NULL,
  `table_k` BIGINT UNSIGNED NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB;
\end{verbatim}

\subsection{\texttt{facility}、\texttt{cubby}、\texttt{slot\_snapshot}、\texttt{tier\_stat}}
\begin{verbatim}
CREATE TABLE IF NOT EXISTS `facility`(
  `id` INT NOT NULL,
  `snapshot_id` BIGINT NOT NULL,
  `tail_cubby_id` INT NOT NULL,
  PRIMARY KEY (`id`,`snapshot_id`),
  KEY `ix_fac_snap` (`snapshot_id`)
) ENGINE=InnoDB;

CREATE TABLE IF NOT EXISTS `cubby`(
  `id` INT NOT NULL,
  `snapshot_id` BIGINT NOT NULL,
  `facility_id` INT NOT NULL,
  `tier` INT NOT NULL,
  `capacity` INT NOT NULL,
  `size` INT NOT NULL,
  `is_tail` TINYINT(1) NOT NULL,
  PRIMARY KEY (`id`,`snapshot_id`),
  KEY `ix_cub_snap` (`snapshot_id`)
) ENGINE=InnoDB;

CREATE TABLE IF NOT EXISTS `slot_snapshot`(
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `snapshot_id` BIGINT NOT NULL,
  `cubby_id` INT NOT NULL,
  `slot_index` INT NOT NULL,
  `occupied` TINYINT(1) NOT NULL,
  `key_hash` BIGINT UNSIGNED NULL,
  `probe_length` INT NOT NULL,
  PRIMARY KEY (`id`),
  KEY `ix_slot_snap` (`snapshot_id`),
  KEY `ix_slot_cubby` (`snapshot_id`,`cubby_id`)
) ENGINE=InnoDB;

CREATE TABLE IF NOT EXISTS `tier_stat`(
  `snapshot_id` BIGINT NOT NULL,
  `facility_id` INT NOT NULL,
  `tier` INT NOT NULL,
  `cubby_count` INT NOT NULL,
  PRIMARY KEY (`snapshot_id`,`facility_id`,`tier`)
) ENGINE=InnoDB;
\end{verbatim}

\subsection{\texttt{otsh\_op\_metrics}（操作级事实表）}
\begin{verbatim}
CREATE TABLE IF NOT EXISTS `otsh_op_metrics`(
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `snapshot_id` BIGINT NOT NULL,
  `operation_type` ENUM('insert','query','delete') NOT NULL,
  `probe_count` INT NOT NULL,
  `kick_count` INT NOT NULL,
  `latency_ns` BIGINT NOT NULL,
  `cubby_tier` INT NOT NULL,
  `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`id`),
  KEY `ix_met_snap` (`snapshot_id`)
) ENGINE=InnoDB;
\end{verbatim}
实现中缓冲满 \texttt{1000} 条或强制 flush 时批量插入，单条 \texttt{INSERT} 语句最多拼接 \texttt{500} 行值。

\subsection{\texttt{otsh\_test\_session\_log}}
供 \texttt{otsh\_tests} 在跑完回归后追加一行 JSON：
\begin{verbatim}
CREATE TABLE IF NOT EXISTS `otsh_test_session_log`(
  `id` BIGINT NOT NULL AUTO_INCREMENT,
  `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  `body` MEDIUMTEXT NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB;
\end{verbatim}

\section{清空语义}
\texttt{/api/init} 在 \texttt{reset=true} 时调用 \texttt{DELETE} 清空上述分析表及主键表（对不存在的历史表名忽略错误），以便重复实验。

\section{运行配置（环境变量）}
\begin{itemize}
  \item 服务：\texttt{BIND\_HOST}、\texttt{PORT}、\texttt{LOG\_FILE}；
  \item 自动化测试 \texttt{otsh\_tests}：\texttt{OTSH\_MYSQL\_HOST}、\texttt{OTSH\_MYSQL\_PORT}、\texttt{OTSH\_MYSQL\_USER}、\texttt{OTSH\_MYSQL\_PASSWORD}、\texttt{OTSH\_MYSQL\_DATABASE}、\texttt{OTSH\_MYSQL\_TABLE}、\texttt{OTSH\_TEST\_LOG}；
  \item 基准 \texttt{otsh\_perf}：\texttt{OTSH\_SEED} 等（见 \texttt{perf/bench.cpp} 注释）。
\end{itemize}
